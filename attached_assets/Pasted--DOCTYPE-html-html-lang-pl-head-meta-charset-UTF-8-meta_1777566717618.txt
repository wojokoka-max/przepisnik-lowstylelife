function init(){
  renderQuickCats();
  document.addEventListener('click', e=>{
    if(!e.target.closest('.search-wrap')) closeDropdown();
    // Przyciski w wyniku
    const btn = e.target.closest('button[data-action]');
    if(!btn) return;
    if(btn.dataset.action==='reset') clearSelected();
    if(btn.dataset.action==='variant') createMeal();
  });
  document.getElementById('searchInput').addEventListener('keydown', onKey);
}

function renderQuickCats(){
  document.getElementById('quickCats').innerHTML = QUICK_CATS.map(c=>`
    <div class="qcat" onclick="showCatDropdown('${c}', event)">${CAT_LABELS[c]}</div>
  `).join('');
}

function onSearch(){
  const q = document.getElementById('searchInput').value.trim().toLowerCase();
  if(!q){closeDropdown();return;}
  const matches = DB.filter(p=>p.name.toLowerCase().includes(q)).slice(0,10);
  renderDropdown(matches);
}
function onSearchFocus(){ onSearch(); }

function showCatDropdown(cat, event){
  if(event) event.stopPropagation();
  document.getElementById('searchInput').value='';
  const items = DB.filter(p=>p.cat===cat);
  renderDropdown(items);
}

function renderDropdown(items){
  const dd = document.getElementById('dropdown');
  if(!items.length){closeDropdown();return;}
  hiIdx=-1;
  dd.innerHTML = items.map(p=>`
    <div class="drop-item" data-id="${p.id}" onclick="selectProduct('${p.id}')">
      <span class="drop-item-icon">${CAT_ICONS[p.cat]||'•'}</span>
      <div>
        <div class="drop-item-name">${p.name}</div>
        <div class="drop-item-cat">${CAT_LABELS[p.cat]||p.cat}</div>
      </div>
    </div>
  `).join('');
  dd.classList.add('open');
}

function closeDropdown(){
  const dd=document.getElementById('dropdown');
  dd.classList.remove('open');
  dd.innerHTML='';
}

function onKey(e){
  const dd=document.getElementById('dropdown');
  const items=dd.querySelectorAll('.drop-item');
  if(!dd.classList.contains('open')||!items.length) return;
  if(e.key==='ArrowDown'){e.preventDefault();hiIdx=Math.min(hiIdx+1,items.length-1);items.forEach((el,i)=>el.classList.toggle('hi',i===hiIdx));}
  else if(e.key==='ArrowUp'){e.preventDefault();hiIdx=Math.max(hiIdx-1,0);items.forEach((el,i)=>el.classList.toggle('hi',i===hiIdx));}
  else if(e.key==='Enter'&&hiIdx>=0){e.preventDefault();selectProduct(items[hiIdx].dataset.id);}
  else if(e.key==='Escape') closeDropdown();
}

let _lastRecipeId = null;

function findRecipe(p){
  const matches = RECIPES.filter(r=>r.id!=='fallback' && r.match(p));
  if(!matches.length) return RECIPES.find(r=>r.id==='fallback');
  if(matches.length === 1) return matches[0];
  const pool = matches.filter(r=>r.id !== _lastRecipeId);
  const pick = pool.length ? pool[Math.floor(Math.random() * pool.length)] : matches[Math.floor(Math.random() * matches.length)];
  _lastRecipeId = pick.id;
  return pick;
}

function resolveIngredients(recipe, starProduct){
  const list = [];
  // Składnik gwiazdkowy — defaultowa gramaturia
  const starG = getDefaultG(starProduct);
  list.push({product:starProduct, g:starG, isStar:true});

  // Dodatkowe składniki z przepisu
  const addList = typeof recipe.addIngredients === 'function'
    ? recipe.addIngredients(starProduct)
    : recipe.addIngredients;

  addList.forEach(item=>{
    if(item.ref){
      const p = DB.find(x=>x.id===item.ref);
      if(p) list.push({product:p, g:item.g, isStar:false});
    } else if(item.custom){
      list.push({product:{id:'custom_'+Math.random(),name:item.custom,cat:item.cat||'dodatki'}, g:item.g, isStar:false, isCustom:true});
    }
  });
  return list;
}

function getDefaultG(p){
  if(p.cat==='tluszcze'||p.cat==='dodatki') return 15;
  if(p.cat==='orzechy') return 30;
  if(p.cat==='czekolada') return 25;
  if(p.cat==='sery') return 60;
  if(p.id==='jajko') return 220; // ~2 jajka
  if(p.cat==='nabial') return 150;
  if(p.cat==='mieso'||p.cat==='ryby') return 150;
  if(p.cat==='owoce') return 100;
  return 150;
}

function createMeal(){
  if(!starProduct) return;

  const recipe = findRecipe(starProduct);
  const ingredients = resolveIngredients(recipe, starProduct);
  const steps = recipe.steps(starProduct);
  const title = recipe.name(starProduct);
  const tip = recipe.tip || 'Smacznego!';

  // Loading animation
  const loading = document.getElementById('loading');
  const box = document.getElementById('resultBox');
  loading.classList.add('visible');
  box.classList.remove('visible');
  box.innerHTML='';

  setTimeout(()=>{
    loading.classList.remove('visible');

    box.innerHTML = `
      <div class="result-header">
        <div class="result-type">${recipe.type}</div>
        <div class="result-title">${title}</div>
        <div class="result-time">⏱ ok. ${recipe.time} minut</div>
      </div>
      <div class="result-body">

        <div class="section-title">Składniki</div>
        <div class="ing-grid">
          ${ingredients.map(ing=>`
            <div class="ing-card ${ing.isStar?'star':''}">
              <span class="ing-card-icon">${CAT_ICONS[ing.product.cat]||'•'}</span>
              <div class="ing-card-info">
                <div class="ing-card-name">${ing.product.name}${ing.isStar?' ★':''}</div>
                <div class="ing-card-g">${ing.g}g</div>
              </div>
            </div>
          `).join('')}
        </div>

        <div class="section-title">Przygotowanie</div>
        <div class="steps">
          ${steps.map((s,i)=>`
            <div class="step" style="animation-delay:${i*0.07}s">
              <div class="step-num">${i+1}</div>
              <div class="step-text">${s}</div>
            </div>
          `).join('')}
        </div>

        <div class="tip">
          <div class="tip-head">💡 Wskazówka</div>
          <div class="tip-text">${tip}</div>
        </div>

        <div class="actions">
          <button class="btn-sec" data-action="reset">← Zmień składnik</button>
          <button class="btn-grn" data-action="variant">↻ Inny wariant</button>
        </div>
      </div>
    `;
    box.classList.add('visible');
    setTimeout(()=>box.scrollIntoView({behavior:'smooth',block:'start'}), 80);
  }, 700);
}
